name: TruffleHog Secret Scan Module

on:
  workflow_call:

jobs:
  secret-scan:
    runs-on: self-hosted

    steps:
      # 1. Checkout full history
      - name: Checkout Source Code
        uses: actions/checkout@v3
        with:
          fetch-depth: 0

      # 2. Prepare variables
      - name: Prepare Scan Variables
        run: |
          echo "Preparing secret scanning environment"
          echo "SCAN_TARGET=." >> $GITHUB_ENV
          echo "RESULT_FILE=scan-output.json" >> $GITHUB_ENV

      # 3. Run TruffleHog scan
      - name: Execute TruffleHog Scan
        run: |
          echo "Running TruffleHog scan"

          trufflehog filesystem "$SCAN_TARGET" \
            --json \
            --no-update \
            > "$RESULT_FILE" 2>&1

          echo "Scan completed"

        - name: Extract Secret Details
          run: |
            echo "==== Detected Secrets ===="
            if [ -f scan-output.json ]; then
              grep -E '"DetectorName"|"Raw"|"file"' scan-output.json || echo "No secrets found"
            else
              echo "No scan output file"
            fi

      # 4. Ensure file always exists (important)
      - name: Ensure Result File Exists
        run: |
          if [ ! -f "$RESULT_FILE" ]; then
            echo "{}" > "$RESULT_FILE"
          fi

      # 5. Evaluate results
      - name: Evaluate Scan Output
        run: |
          echo "Evaluating scan results"

          if [ -s "$RESULT_FILE" ]; then
            SECRET_TOTAL=$(grep -c "DetectorName" "$RESULT_FILE" || echo "0")
          else
            SECRET_TOTAL=0
          fi

          echo "Secrets detected: $SECRET_TOTAL"

          if [ "$SECRET_TOTAL" -gt 0 ]; then
            echo "SECRET_FOUND=true" >> $GITHUB_ENV
            exit 1
          else
            echo "SECRET_FOUND=false" >> $GITHUB_ENV
            echo "Secret scan passed"
          fi

      # 6. Debug (optional but useful)
      - name: Debug - List Files
        run: |
          echo "Listing files in workspace"
          ls -la

      # 7. PR Comment (FIXED)
      - name: Comment Scan Results
        if: github.event_name == 'pull_request'
        uses: actions/github-script@v7
        with:
          script: |
            const fs = require('fs');

            const path = 'scan-output.json';
            let message = "### 🔐 TruffleHog Secret Scan Result\n\n";

            if (fs.existsSync(path)) {
              const content = fs.readFileSync(path, 'utf8');

              if (content.includes("DetectorName")) {
                message += "⚠️ Potential secrets detected. Please review workflow logs.\n";
              } else {
                message += "✅ No secrets detected.\n";
              }
            } else {
              message += "⚠️ Scan result file not found.\n";
            }

            github.rest.issues.createComment({
              issue_number: context.issue.number,
              owner: context.repo.owner,
              repo: context.repo.repo,
              body: message
            });
