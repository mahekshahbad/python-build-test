name: TruffleHog Secret Scan

on:
  workflow_call:
    inputs:
      working-directory:
        required: false
        type: string
        default: "."
      fail-on-secrets:
        required: false
        type: boolean
        default: true
      upload-artifact:
        required: false
        type: boolean
        default: true

jobs:
  trufflehog-scan:
    runs-on: self-hosted

    steps:
      # ---------------------------
      # 1. Checkout
      # ---------------------------
      - name: Checkout Code
        uses: actions/checkout@v3
        with:
          fetch-depth: 0

      # ---------------------------
      # 2. Prepare Ignore File
      # ---------------------------
      - name: Prepare TruffleHog Ignore Rules
        id: ignore
        shell: bash
        run: |
          echo "Preparing ignore rules..."

          TRUFFLEHOG_IGNORE="${RUNNER_TEMP}/.trufflehogignore"

          printf "%s\n" \
            ".git/" \
            "node_modules/" \
            ".terraform/" \
            ".terraform.lock.hcl" \
            ".gitmodules" \
            "test-results/" \
            "security-reports/" \
            "*.sarif" \
            "*.json" \
            > "$TRUFFLEHOG_IGNORE"

          if [ -f ".trufflehogignore" ]; then
            echo "Merging repo ignore file..."
            cat ".trufflehogignore" >> "$TRUFFLEHOG_IGNORE"
          fi

          echo "IGNORE_FILE=$TRUFFLEHOG_IGNORE" >> $GITHUB_ENV

      # ---------------------------
      # 3. Run TruffleHog Scan
      # ---------------------------
      - name: Run TruffleHog Scan
        id: scan
        shell: bash
        working-directory: ${{ inputs.working-directory }}
        continue-on-error: true
        run: |
          echo "::group::Running TruffleHog Scan"

          trufflehog filesystem . \
            --json \
            --no-update \
            --exclude-paths="$IGNORE_FILE" \
            > trufflehog-results.json 2>&1 || true

          echo "::endgroup::"

      # ---------------------------
      # 4. Analyze Results
      # ---------------------------
      - name: Analyze Scan Results
        id: analyze
        shell: bash
        working-directory: ${{ inputs.working-directory }}
        run: |
          echo "::group::Analyzing Results"

          if [[ -s trufflehog-results.json ]]; then
            SECRET_COUNT=$(grep -c '"DetectorName"' trufflehog-results.json || echo "0")
            SECRET_COUNT=$(echo "$SECRET_COUNT" | tr -d '[:space:]' | grep -o '[0-9]*' | head -1)
            SECRET_COUNT=${SECRET_COUNT:-0}
          else
            SECRET_COUNT=0
          fi

          echo "SECRET_COUNT=$SECRET_COUNT" >> $GITHUB_ENV
          echo "Secrets detected: $SECRET_COUNT"

          if [[ "$SECRET_COUNT" -gt 0 ]]; then
            echo "::error::TruffleHog found $SECRET_COUNT potential secret(s)"

            echo "---- Secret Details ----"
            grep '"DetectorName"' trufflehog-results.json | head -20 || true
          else
            echo "No secrets detected"
          fi

          echo "::endgroup::"

      # ---------------------------
      # 5. Convert to SARIF
      # ---------------------------
      - name: Convert to SARIF
        if: always()
        shell: bash
        working-directory: ${{ inputs.working-directory }}
        run: |
          echo "::group::Converting to SARIF"

          if [[ ! -s trufflehog-results.json ]]; then
            echo "No results to convert"
            exit 0
          fi

          jq -s '
          {
            "version": "2.1.0",
            "runs": [{
              "tool": {
                "driver": {
                  "name": "TruffleHog",
                  "informationUri": "https://github.com/trufflesecurity/trufflehog"
                }
              },
              "results": map({
                "ruleId": .DetectorName,
                "message": {
                  "text": ("Potential secret detected: " + .DetectorName)
                },
                "level": "error"
              })
            }]
          }' trufflehog-results.json > trufflehog-results.sarif || echo "{}" > trufflehog-results.sarif

          echo "::endgroup::"

      # ---------------------------
      # 6. Upload Artifacts
      # ---------------------------
      - name: Upload Scan Artifacts
        if: ${{ inputs.upload-artifact }}
        uses: actions/upload-artifact@v4
        with:
          name: trufflehog-results
          path: |
            ${{ inputs.working-directory }}/trufflehog-results.json
            ${{ inputs.working-directory }}/trufflehog-results.sarif

      # ---------------------------
      # 7. Comment on PR
      # ---------------------------
      - name: Comment on PR
        if: github.event_name == 'pull_request'
        uses: actions/github-script@v7
        with:
          script: |
            const count = process.env.SECRET_COUNT || "0";
            let message = `### 🔐 TruffleHog Scan Result\n\n`;

            if (count > 0) {
              message += `❌ Found ${count} potential secret(s). Please check workflow logs.\n`;
            } else {
              message += `✅ No secrets detected.\n`;
            }

            github.rest.issues.createComment({
              issue_number: context.issue.number,
              owner: context.repo.owner,
              repo: context.repo.repo,
              body: message
            });

      # ---------------------------
      # 8. Fail Job (Final Gate)
      # ---------------------------
      - name: Fail if Secrets Found
        if: ${{ inputs.fail-on-secrets }}
        shell: bash
        run: |
          if [[ "$SECRET_COUNT" -gt 0 ]]; then
            echo "Failing due to detected secrets"
            exit 1
          fi
