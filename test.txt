name: TruffleHog Secret Scan

on:
  workflow_call:
    inputs:
      working-directory:
        required: false
        type: string
        default: "."
      fail-on-secrets:
        required: false
        type: boolean
        default: true
      upload-artifact:
        required: false
        type: boolean
        default: true

jobs:
  trufflehog-scan:
    runs-on: self-hosted

    steps:
      # -------------------------
      # 1. Checkout
      # -------------------------
      - name: Checkout Code
        uses: actions/checkout@v3
        with:
          fetch-depth: 0

      # -------------------------
      # 2. Setup Ignore Rules
      # -------------------------
      - name: Prepare TruffleHog Ignore File
        id: ignore
        run: |
          echo "Creating ignore file"

          TRUFFLEHOG_IGNORE="${RUNNER_TEMP}/.trufflehogignore"

          printf "%s\n" \
          ".git/" \
          "node_modules/" \
          ".terraform/" \
          ".terraform.lock.hcl" \
          "test-results/" \
          "security-reports/" \
          ".*\\.sarif$" \
          ".*\\.json$" \
          > "$TRUFFLEHOG_IGNORE"

          echo "TRUFFLEHOG_IGNORE=$TRUFFLEHOG_IGNORE" >> $GITHUB_ENV

      # -------------------------
      # 3. Run TruffleHog Scan
      # -------------------------
      - name: Run TruffleHog Scan
        id: scan
        working-directory: ${{ inputs.working-directory }}
        run: |
          echo "Starting TruffleHog scan..."

          trufflehog filesystem . \
            --json \
            --no-update \
            --exclude-paths="$TRUFFLEHOG_IGNORE" \
            > trufflehog-results.json 2>&1

          echo "Scan completed"

      # -------------------------
      # 4. Evaluate Results
      # -------------------------
      - name: Evaluate Scan Results
        id: evaluate
        working-directory: ${{ inputs.working-directory }}
        run: |
          echo "Evaluating scan results..."

          if [ ! -f trufflehog-results.json ]; then
            echo "❌ Scan output file not found"
            exit 1
          fi

          SECRET_COUNT=$(grep -c '"DetectorName"' trufflehog-results.json || echo "0")
          SECRET_COUNT=$(echo "$SECRET_COUNT" | tr -d '[:space:]')

          echo "Secrets detected: $SECRET_COUNT"

          echo "SECRET_COUNT=$SECRET_COUNT" >> $GITHUB_ENV

          if [ "$SECRET_COUNT" -gt 0 ]; then
            echo "❌ Secrets found!"

            if [ "${{ inputs.fail-on-secrets }}" = "true" ]; then
              exit 1
            fi
          else
            echo "✅ No secrets found"
          fi

      # -------------------------
      # 5. Convert to SARIF
      # -------------------------
      - name: Convert Results to SARIF
        if: always()
        working-directory: ${{ inputs.working-directory }}
        run: |
          echo "Converting results to SARIF..."

          if [ ! -s trufflehog-results.json ]; then
            echo "No results to convert"
            echo '{"version":"2.1.0","runs":[]}' > trufflehog-results.sarif
            exit 0
          fi

          jq -s '
          {
            "version": "2.1.0",
            "runs": [
              {
                "tool": {
                  "driver": {
                    "name": "TruffleHog",
                    "informationUri": "https://github.com/trufflesecurity/trufflehog"
                  }
                },
                "results": (
                  map(select(.DetectorName != null)) | map({
                    "ruleId": .DetectorName,
                    "message": {
                      "text": ("Potential secret detected: " + .DetectorName)
                    },
                    "level": "error",
                    "locations": [
                      {
                        "physicalLocation": {
                          "artifactLocation": {
                            "uri": (.SourceMetadata.Data.Filesystem.file // "unknown")
                          },
                          "region": {
                            "startLine": (.SourceMetadata.Data.Filesystem.line // 1)
                          }
                        }
                      }
                    ]
                  })
                )
              }
            ]
          }
          ' trufflehog-results.json > trufflehog-results.sarif

      # -------------------------
      # 6. Upload Artifact
      # -------------------------
      - name: Upload Results
        if: ${{ inputs.upload-artifact }}
        uses: actions/upload-artifact@v4
        with:
          name: trufflehog-results
          path: |
            ${{ inputs.working-directory }}/trufflehog-results.json
            ${{ inputs.working-directory }}/trufflehog-results.sarif

      # -------------------------
      # 7. PR Comment
      # -------------------------
      - name: Comment on PR
        if: github.event_name == 'pull_request'
        uses: actions/github-script@v7
        with:
          script: |
            const secretCount = process.env.SECRET_COUNT || "0";

            let message = "### 🔐 TruffleHog Scan Result\n\n";

            if (secretCount > 0) {
              message += `❌ ${secretCount} potential secret(s) detected.\n`;
              message += "Please review workflow logs.\n";
            } else {
              message += "✅ No secrets detected.\n";
            }

            github.rest.issues.createComment({
              issue_number: context.issue.number,
              owner: context.repo.owner,
              repo: context.repo.repo,
              body: message
            });

name: TruffleHog Secret Scan

on:
  workflow_call:
    inputs:
      working-directory:
        description: "Directory to scan"
        required: false
        type: string
        default: "."

      fail-on-secrets:
        description: "Fail workflow if secrets found"
        required: false
        type: boolean
        default: true

      scan-path:
        description: "Path to scan (relative)"
        required: false
        type: string
        default: "."

      exclude-paths:
        description: "Custom exclude patterns (newline separated)"
        required: false
        type: string
        default: ""

      upload-artifact:
        description: "Upload scan results"
        required: false
        type: boolean
        default: true

jobs:
  trufflehog-scan:
    runs-on: ubuntu-latest

    steps:
      # -----------------------------
      # 1. Checkout
      # -----------------------------
      - name: Checkout
        uses: actions/checkout@v4

      # -----------------------------
      # 2. Install TruffleHog
      # -----------------------------
      - name: Install TruffleHog
        run: |
          curl -sSfL https://raw.githubusercontent.com/trufflesecurity/trufflehog/main/scripts/install.sh | sh
          echo "$HOME/.local/bin" >> $GITHUB_PATH

      # -----------------------------
      # 3. Prepare Ignore File
      # -----------------------------
      - name: Prepare Ignore File
        run: |
          IGNORE_FILE="${RUNNER_TEMP}/.trufflehogignore"

          printf "%s\n" \
          ".git/" \
          "node_modules/" \
          ".terraform/" \
          ".terraform.lock.hcl" \
          "test-results/" \
          "security-reports/" \
          '\.sarif$' \
          '\.json$' \
          > "$IGNORE_FILE"

          # Add user input excludes
          if [[ -n "${{ inputs.exclude-paths }}" ]]; then
            echo "${{ inputs.exclude-paths }}" >> "$IGNORE_FILE"
          fi

          echo "IGNORE_FILE=$IGNORE_FILE" >> $GITHUB_ENV

      # -----------------------------
      # 4. Run Scan
      # -----------------------------
      - name: Run TruffleHog Scan
        working-directory: ${{ inputs.working-directory }}
        run: |
          echo "Running scan on ${{ inputs.scan-path }}"

          trufflehog filesystem "${{ inputs.scan-path }}" \
            --json \
            --no-update \
            --results=verified,unknown \
            --exclude-paths="$IGNORE_FILE" \
            > trufflehog-results.json

      # -----------------------------
      # 5. Debug (important for you)
      # -----------------------------
      - name: Debug Output
        run: |
          echo "==== FIRST 20 LINES ===="
          head -n 20 trufflehog-results.json || true

      # -----------------------------
      # 6. Evaluate Results
      # -----------------------------
      - name: Evaluate Results
        id: evaluate
        run: |
          if [[ -s trufflehog-results.json ]]; then
            SECRET_COUNT=$(jq -s 'map(select(.DetectorName != null)) | length' trufflehog-results.json)
          else
            SECRET_COUNT=0
          fi

          echo "Secrets detected: $SECRET_COUNT"

          if [[ "$SECRET_COUNT" -gt 0 ]]; then
            echo "::error::Secrets found: $SECRET_COUNT"

            if [[ "${{ inputs.fail-on-secrets }}" == "true" ]]; then
              exit 1
            fi
          else
            echo "No secrets found"
          fi

      # -----------------------------
      # 7. Convert to SARIF
      # -----------------------------
      - name: Convert to SARIF
        if: always()
        run: |
          if [[ ! -s trufflehog-results.json ]]; then
            echo '{"version":"2.1.0","runs":[]}' > trufflehog-results.sarif
            exit 0
          fi

          jq -s '
          {
            "version": "2.1.0",
            "runs": [
              {
                "tool": {
                  "driver": {
                    "name": "TruffleHog",
                    "informationUri": "https://github.com/trufflesecurity/trufflehog"
                  }
                },
                "results": [
                  .[] | select(.DetectorName != null) | {
                    ruleId: (.DetectorName // "unknown"),
                    message: {
                      text: ("Secret detected: " + (.DetectorName // "unknown"))
                    },
                    level: "error"
                  }
                ]
              }
            ]
          }' trufflehog-results.json > trufflehog-results.sarif

      # -----------------------------
      # 8. Upload Results
      # -----------------------------
      - name: Upload Results
        if: ${{ inputs.upload-artifact }}
        uses: actions/upload-artifact@v4
        with:
          name: trufflehog-results
          path: |
            trufflehog-results.json
            trufflehog-results.sarif
