on:
  push:
    branches:
      - master

jobs:
  deploy:
    runs-on: self-hosted

    env:
      ACR_LOGIN_SERVER: ${{ secrets.ACR_LOGIN_SERVER_SANDBOX }}
      IMAGE_NAME: sds-codeset-extract
      NAMESPACE: sds-codeset-extract-dev

    steps:
      - name: Checkout
        uses: actions/checkout@v4

      # Azure Login
      - name: Azure Login
        uses: azure/login@v3
        with:
          client-id: ${{ secrets.AZURE_CLIENT_ID }}
          tenant-id: ${{ secrets.AZURE_TENANT_ID }}
          subscription-id: ${{ secrets.AZURE_SUBSCRIPTION_ID }}

      # ACR Login
      - name: ACR Login
        run: az acr login --name $ACR_LOGIN_SERVER

      # Prepare Image Tag 
      - name: Prepare Docker Metadata
        id: docker_meta
        run: |
          DATE=$(date +%Y%m%d)
          GIT_SHA=${GITHUB_SHA::8}
          BUILD_NUMBER=${GITHUB_RUN_NUMBER}
          BRANCH_NAME=${GITHUB_REF_NAME}

          VERSION="${DATE}.${GIT_SHA}.${BUILD_NUMBER}"
          IMAGE_TAG="${VERSION}-${BRANCH_NAME}"

          echo "IMAGE_TAG=$IMAGE_TAG" >> $GITHUB_OUTPUT

      # Docker Build & Push 
      - name: Docker Build and Push
        uses: medica-dev-platform/github-common-actions/.github/actions/docker-build@feature/maven-build-action
        with:
          image-name: ${{ env.ACR_LOGIN_SERVER }}/${{ env.IMAGE_NAME }}
          image-tag: ${{ steps.docker_meta.outputs.IMAGE_TAG }}
          dockerfile-path: ./Dockerfile
          build-context: .
          push: true

      # Set AKS context
      - name: Set AKS Context
        uses: azure/aks-set-context@v5
        with:
          cluster-name: k8s-20-npe-cicd-02
          resource-group: rg-20-npe-k8s-01

      # Create Namespace
      - name: Create Namespace if not exists
        run: |
          kubectl get namespace $NAMESPACE || kubectl create namespace $NAMESPACE

      # Deploy to AKS
      - name: Deploy to AKS
        run: |
          IMAGE="${{ env.ACR_LOGIN_SERVER }}/${{ env.IMAGE_NAME }}:${{ steps.docker_meta.outputs.IMAGE_TAG }}"

          cat <<EOF | kubectl apply -f -
          apiVersion: apps/v1
          kind: Deployment
          metadata:
            name: sds-codeset-extract
            namespace: $NAMESPACE
          spec:
            replicas: 1
            selector:
              matchLabels:
                app: sds-codeset-extract
            template:
              metadata:
                labels:
                  app: sds-codeset-extract
              spec:
                containers:
                - name: sds-codeset-extract
                  image: $IMAGE
                  ports:
                  - containerPort: 8000
                  imagePullPolicy: Always
          EOF

      # Verify Deployment
      - name: Verify Deployment
        run: |
          kubectl rollout status deployment/sds-codeset-extract -n $NAMESPACE

name: TruffleHog Secret Scan

on:
  workflow_call:
    inputs:
      working-directory:
        required: false
        type: string
        default: "."
      fail-on-secrets:
        required: false
        type: boolean
        default: true
      upload-artifact:
        required: false
        type: boolean
        default: true
      trufflehog-ignore-patterns:
        description: "Custom ignore patterns"
        required: false
        type: string
        default: ""
      scan-issues-prs:
        description: "Scan GitHub issues and PR comments"
        required: false
        type: boolean
        default: true

permissions:
  contents: read
  issues: read
  pull-requests: read

jobs:
  trufflehog-scan:
    runs-on: self-hosted

    steps:
      # -------------------------
      # 1. Checkout
      # -------------------------
      - name: Checkout Code
        uses: actions/checkout@v3
        with:
          fetch-depth: 0

      # -------------------------
      # 2. Setup Ignore Rules
      # -------------------------
      - name: Prepare TruffleHog Ignore File
        id: ignore
        run: |
          echo "Creating ignore file"
      
          TRUFFLEHOG_IGNORE="${RUNNER_TEMP}/.trufflehogignore"
      
          # Default patterns (safe baseline)
          printf "%s\n" \
            ".git/" \
            "node_modules/" \
            ".terraform/" \
            ".terraform.lock.hcl" \
            "test-results/" \
            "security-reports/" \
            ".*\\.sarif$" \
            ".*\\.json$" \
            > "$TRUFFLEHOG_IGNORE"
      
          # Append user-provided patterns
          if [ -n "${{ inputs.trufflehog-ignore-patterns }}" ]; then
            echo "Adding user provided ignore patterns"
            echo "${{ inputs.trufflehog-ignore-patterns }}" >> "$TRUFFLEHOG_IGNORE"
          fi
      
          echo "TRUFFLEHOG_IGNORE=$TRUFFLEHOG_IGNORE" >> $GITHUB_ENV


      # -------------------------
      # 3. Run TruffleHog Scan
      # -------------------------
      - name: Run TruffleHog Scan
        id: scan
        working-directory: ${{ inputs.working-directory }}
        run: |
          echo "Starting TruffleHog scan..."

          trufflehog filesystem . \
            --json \
            --no-update \
            --exclude-paths="$TRUFFLEHOG_IGNORE" \
            > trufflehog-results.json 2>&1 || true

          echo "Scan completed"


      # --------------------------
      # 3B. Scan Issues & PR Comments
      # --------------------------
      - name: Scan GitHub Issues & PRs
        if: ${{ inputs.scan-issues-prs }}
        env:
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
        run: |
          echo "Scanning GitHub issues and PR comments..."
      
          trufflehog github \
            --repo="${{ github.repository }}" \
            --json \
            --no-update \
            > trufflehog-github-results.json 2>&1 || true
      
          echo "GitHub scan completed"


      # --------------------------
      # Merge Results
      # --------------------------
      - name: Merge Scan Results
        run: |
          echo "Merging scan results..."
      
          if [ -f trufflehog-github-results.json ]; then
            cat trufflehog-results.json trufflehog-github-results.json > combined-results.json
            mv combined-results.json trufflehog-results.json
          fi

      # -------------------------
      # 4. Evaluate Results
      # -------------------------
      - name: Evaluate Scan Results
        id: evaluate
        working-directory: ${{ inputs.working-directory }}
        run: |
          echo "Evaluating scan results..."

          if [ ! -f trufflehog-results.json ]; then
            echo "❌ Scan output file not found"
            exit 1
          fi

          SECRET_COUNT=$(grep -c '"DetectorName"' trufflehog-results.json || echo "0")
          SECRET_COUNT=$(echo "$SECRET_COUNT" | tr -d '[:space:]')

          echo "Secrets detected: $SECRET_COUNT"

          echo "SECRET_COUNT=$SECRET_COUNT" >> $GITHUB_ENV

          if [ "$SECRET_COUNT" -gt 0 ]; then
            echo "❌ Secrets found!"
            echo "---- Detected Secrets ----"
          
            jq -r '
            select(.DetectorName != null) |
            "Detector: \(.DetectorName) |
             Source: \(
               .SourceMetadata.Data.Filesystem.file
               // (.SourceMetadata.Data.GitHub.repository + " (Issue/PR)")
               // "unknown"
             ) |
             Line: \(
               .SourceMetadata.Data.Filesystem.line
               // "N/A"
             )"
          ' trufflehog-results.json
          
            echo "--------------------------"
          
            if [ "${{ inputs.fail-on-secrets }}" = "true" ]; then
              exit 1
            fi
          fi

      # -------------------------
      # 5. Convert to SARIF
      # -------------------------
      - name: Convert Results to SARIF
        if: always()
        working-directory: ${{ inputs.working-directory }}
        run: |
          echo "Converting results to SARIF..."

          if [ ! -s trufflehog-results.json ]; then
            echo "No results to convert"
            echo '{"version":"2.1.0","runs":[]}' > trufflehog-results.sarif
            exit 0
          fi

          jq -s '
          {
            "version": "2.1.0",
            "runs": [
              {
                "tool": {
                  "driver": {
                    "name": "TruffleHog",
                    "informationUri": "https://github.com/trufflesecurity/trufflehog"
                  }
                },
                "results": (
                  map(select(.DetectorName != null)) | map({
                    "ruleId": .DetectorName,
                    "message": {
                      "text": ("Potential secret detected: " + .DetectorName)
                    },
                    "level": "error",
                    "locations": [
                      {
                        "physicalLocation": {
                          "artifactLocation": {
                            "uri": (.SourceMetadata.Data.Filesystem.file // "unknown")
                          },
                          "region": {
                            "startLine": (.SourceMetadata.Data.Filesystem.line // 1)
                          }
                        }
                      }
                    ]
                  })
                )
              }
            ]
          }
          ' trufflehog-results.json > trufflehog-results.sarif

      # -------------------------
      # 6. Upload Artifact
      # -------------------------
      - name: Upload Results
        if: ${{ always() && inputs.upload-artifact }}
        uses: actions/upload-artifact@v4
        with:
          name: trufflehog-results
          path: |
            ${{ inputs.working-directory }}/trufflehog-results.json
            ${{ inputs.working-directory }}/trufflehog-results.sarif

      # -------------------------
      # 7. PR Comment
      # -------------------------
      - name: Comment on PR
        if: ${{ always() && github.event_name == 'pull_request' }}
        uses: actions/github-script@v7
        with:
          script: |
            const secretCount = process.env.SECRET_COUNT || "0";

            let message = "### 🔐 TruffleHog Scan Result\n\n";

            if (secretCount > 0) {
              message += `❌ ${secretCount} potential secret(s) detected.\n`;
              message += "Please review workflow logs.\n";
            } else {
              message += "✅ No secrets detected.\n";
            }

            github.rest.issues.createComment({
              issue_number: context.issue.number,
              owner: context.repo.owner,
              repo: context.repo.repo,
              body: message
            });
